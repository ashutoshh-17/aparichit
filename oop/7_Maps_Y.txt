#include <iostream>
#include <map>
#include <string>

int main() {
    // Creating a map to store state populations
    std::map<std::string, long long> statePopulations;

    // Adding state-population pairs to the map
    statePopulations["Uttar Pradesh"] = 220000000;
    statePopulations["Maharashtra"] = 124000000;
    statePopulations["Bihar"] = 120000000;
    statePopulations["West Bengal"] = 99000000;
    statePopulations["Madhya Pradesh"] = 78000000;
    statePopulations["Tamil Nadu"] = 76000000;
    statePopulations["Rajasthan"] = 74000000;
    statePopulations["Karnataka"] = 67000000;
    statePopulations["Gujarat"] = 64000000;
    statePopulations["Andhra Pradesh"] = 53000000;

    // Prompting the user to enter a state name
    std::string inputState;
    std::cout << "Enter the name of a state in India: ";
    std::getline(std::cin, inputState);

    // Looking up and displaying the population of the entered state
    auto it = statePopulations.find(inputState);
    if (it != statePopulations.end()) {
        std::cout << "Population of " << inputState << ":" << it->second << std::endl;
    } else {
        std::cout << "State not found in the map." << std::endl;
    }

    return 0;
}
